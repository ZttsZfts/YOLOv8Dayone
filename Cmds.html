单卡训练 
yolo task=detect mode=train model=yolov8n.pt data=mark1.yaml batch=32 epochs=100 imgsz=640 workers=16 device=0
模型验证 
yolo task=detect mode=val model=runs/detect/train/weights/best.pt data=mark1.yaml device=0
模型预测（调用摄像机） 
yolo task=detect mode=predict model=runs/detect/train/weights/best.pt source=0 device=0
模型导出onnx 
yolo task=detect mode=export model=runs/detect/train/weights/best.pt format=onnx
